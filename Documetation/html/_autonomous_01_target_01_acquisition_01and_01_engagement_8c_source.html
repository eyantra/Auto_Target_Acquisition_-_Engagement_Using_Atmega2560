00013  There are two components to the motion control:

00018  
00019  Pins for PWM are kept at logic 1.
00020  
00021  Connection Details:     
00022                         
00023   Motion control:        L-1---->PA0;        L-2---->PA1;
00024                            R-1---->PA2;        R-2---->PA3;
00025                            PL3 (OC5A) ----> Logic 1;     PL4 (OC5B) ----> Logic 1;
00026 
00027 
00028   Serial Communication:    PORTD 2 --> RXD1 UART1 receive for RS232 serial communication
00029                         PORTD 3 --> TXD1 UART1 transmit for RS232 serial communication
00030 
00031                         PORTH 0 --> RXD2 UART 2 receive for USB - RS232 communication
00032                         PORTH 1 --> TXD2 UART 2 transmit for USB - RS232 communication
00033 
00034                         PORTE 0 --> RXD0 UART0 receive for ZigBee wireless communication
00035                         PORTE 1 --> TXD0 UART0 transmit for ZigBee wireless communication
00036 
00037                         PORTJ 0 --> RXD3 UART3 receive available on microcontroller expainsion board
00038                         PORTJ 1 --> TXD3 UART3 transmit available on microcontroller expainsion board
00039 
00040 Serial communication baud rate: 9600bps
00041 To control robot use number pad of the keyboard which is located on the right hand side of the keyboard.
00042 Make sure that NUM lock is on.

00050                 5                0x35    Stop
00051                 7                0x37    Buzzer on
00052                 9                0x39    Buzzer off

00186 void motion_set (unsigned char Direction)
00187 {
00188  unsigned char PortARestore = 0;
00189 
00190  Direction &= 0x0F;                     // removing upper nibbel as it is not needed
00191  PortARestore = PORTA;                  // reading the PORTA's original status
00192  PortARestore &= 0xF0;                  // setting lower direction nibbel to 0
00193  PortARestore |= Direction;     // adding lower nibbel for direction command and restoring the PORTA status
00194  PORTA = PortARestore;                  // setting the command to the port
00195 }
