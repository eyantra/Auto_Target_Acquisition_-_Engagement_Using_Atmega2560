00027 
00028   Serial Communication:    PORTD 2 --> RXD1 UART1 receive for RS232 serial communication
00029                         PORTD 3 --> TXD1 UART1 transmit for RS232 serial communication
00030 
00031                         PORTH 0 --> RXD2 UART 2 receive for USB - RS232 communication
00032                         PORTH 1 --> TXD2 UART 2 transmit for USB - RS232 communication
00033

00038                         PORTJ 1 --> TXD3 UART3 transmit available on microcontroller expainsion board

00173 void motion_set (unsigned char Direction)
00174 {
00175  unsigned char PortARestore = 0;
00176 
00177  Direction &= 0x0F;                     // removing upper nibbel as it is not needed
00178  PortARestore = PORTA;                  // reading the PORTA's original status
00179  PortARestore &= 0xF0;                  // setting lower direction nibbel to 0
00180  PortARestore |= Direction;     // adding lower nibbel for direction command and restoring the PORTA status
00181  PORTA = PortARestore;                  // setting the command to the port
00182 }
